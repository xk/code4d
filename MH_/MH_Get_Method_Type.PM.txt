  //@use:  // MH_Get_Method_Type ($MethodName_t:txt) -> $MethodType_t:txtIf (False)  //=================  //@FUNCTION: MH_Get_Method_Type  //@brief: Returns the method type according to the method name  //=================  //@author: jpupier  //@modified_by:  //---------  //@version: 1.0  //@date: 22 oct, 2010  //=================  //@SVN_Folder: MH_/Else   //=================C_TEXT($1;$MethodName_t)$MethodName_t:=$1C_TEXT($0;$MethodType_t)$MethodType_t:=""C_LONGINT($CountDots_l)  //---------C_LONGINT($ParamCount_l)$ParamCount_l:=Count parameters  //  //=================End if   // ============= [Try to determine the method type ] ==============$MethodType_t:=""$CountDots_l:=Length($MethodName_t)-Length(Replace string($MethodName_t;".";""))  //We assume there is no "." in a form nameCase of : ($MethodName_t="Trigger @")  // Trigger  : "Table trigger"$MethodType_t:="TRIG": ($MethodName_t="Compiler@")  // Compiler_ : "Compiler method"$MethodType_t:="COMP": ($MethodName_t="[@") & ($CountDots_l=1)  // [Table].form1 : "Table Form method"$MethodType_t:="TFM": ($MethodName_t="[@") & ($CountDots_l>1)  // [Table].form1.object  : "Table Form Object Method"$MethodType_t:="TFOM": ($CountDots_l>0)  // Form1.object : "Project Form Object Method" $MethodType_t:="PFOM": ($CountDots_l=0)  // Method1  "Project Method" or "Project Form Method" $MethodType_t:="PM"Else   //TRACEEnd case   //=================$0:=$MethodType_t
