  // ----------------------------------------------------  // Nom utilisateur (OS) : Pascal Foubert  // Date et heure : 20/09/10, 15:24:39  // ----------------------------------------------------  // Méthode : Util_Mac_TempFolder_Create  // Description  //    Create a temporary folder in /tmp (which is a symlink to /private/tmp  //    On mac osx the tmp folders are created using a template   //    Every XXXXX are replaced by random chars  //    ex MYDIR.XXXX might become MYDIR.AyUX or MYDIR.ziSD ....  //  // Paramètres  // $1 (Pointer) Point to the new created Temp folder path  // $2 (Text) Template secured temp folder (opt)  // ----------------------------------------------------C_POINTER($1)C_TEXT($2;$0)C_TEXT($Action;$In;$Out;$Error)Case of : (Count parameters=1)$Action:="mktemp -d /tmp/XXX-XXXXX": (Count parameters=2)$Action:="mktemp -d /private/tmp/"+$2+"-XXXXX"End case   //FIXER VARIABLE ENVIRONNEMENT("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"true")LAUNCH EXTERNAL PROCESS($Action;$In;$Out;$Error)$0:=$ErrorIf ($Error="")$1->:=Util_Trim_After ($Out)  // remove blank and \n \tElse ALERT("Impossible de créer le dossier temporaire")End if
