  // ----------------------------------------------------  // User name (OS): Pascal Foubert  // ----------------------------------------------------  // Method: Util_Mac_File_Folder_Untar  //  // Description:  // unTar a tar file on mac osx  //  //  // Parameters  // $1 (Text) Zip full path  // $2 (Text) Destination dir  // $3 (BoolŽen) Asynchrone True/False  //  // Util_Mac_File_Folder_Unzip($ZipFullPath;$DestPath;$Async)  //  // form      : unzip /path/to/zipfile.zip -d /destination/folder/  //  // ----------------------------------------------------C_TEXT($1;$ZipFullPath;$2;$DestPath)C_BOOLEAN($3;$Async)C_TEXT($Action;$In;$Out;$Error)$Async:=FalseIf (Count parameters>=2)  // $ZipFullPath:=$1$DestPath:=$2If (<>System_Path_Separator=":")  // On Mac OS X, should already be = to ":"$Action:="/usr/bin/unzip "$Action:=$Action+"/Volumes/"+Util_Mac_Path_to_Unix ($ZipFullPath)+" -d /Volumes/"+Util_Mac_Path_to_Unix ($DestPath)If (Count parameters=3)$Async:=$3End if If ($Async)SET ENVIRONMENT VARIABLE("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"false")LAUNCH EXTERNAL PROCESS($Action;$In)SET ENVIRONMENT VARIABLE("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"true")$0:=""Else SET ENVIRONMENT VARIABLE("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"true")LAUNCH EXTERNAL PROCESS($Action;$In;$Out;$Error)$0:=$ErrorEnd if End if End if
