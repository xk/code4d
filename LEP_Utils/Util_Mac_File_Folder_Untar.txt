  // ----------------------------------------------------  // User name (OS): Pascal Foubert  // ----------------------------------------------------  // Method: Util_Mac_File_Folder_Untar  //  // Description:  // unTar a tar file on mac osx  //  //  // Parameters  // $1 (Text) Tar full path  // $2 (Text) Destination  // $3 (BoolŽen) Asynchrone True/False  //  // Util_Mac_File_Folder_Untar($TarFullPath;$DestPath;$Async)  //  // form      : tar -C $DestPath -xzpf $TarFullPath  // exemple : tar -C /chemin/vers/destination/extraction/ -xzpf /chemin/vers/tar/filename.tar.gz  //  // ----------------------------------------------------C_TEXT($1;$TarFullPath;$2;$DestPath)C_BOOLEAN($3;$Async)C_TEXT($Action;$In;$Out;$Error)If (Count parameters>=2)  // $TarFullPath:=$1$DestPath:=$2If (<>System_Path_Separator=":")  // On Mac OS X, should already be = to ":"$Action:="/usr/bin/tar -C "$Action:=$Action+"/Volumes/"+Util_Mac_Path_to_Unix ($DestPath)+" -xzpf /Volumes/"+Util_Mac_Path_to_Unix ($TarFullPath)$Async:=$3If ($Async)SET ENVIRONMENT VARIABLE("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"false")LAUNCH EXTERNAL PROCESS($Action;$In)SET ENVIRONMENT VARIABLE("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"true")$0:=""Else SET ENVIRONMENT VARIABLE("_4D_OPTION_BLOCKING_EXTERNAL_PROCESS";"true")LAUNCH EXTERNAL PROCESS($Action;$In;$Out;$Error)$0:=$ErrorEnd if End if End if
