  // ----------------------------------------------------  // Nom utilisateur (OS) : Pascal Foubert  // Date et heure : 16/09/10, 15:07:56  // ----------------------------------------------------  // Mthode : Util_Mac_Pid_Find  //  // Description  // -----------  // Look for the pid of the app set in param.  // If we set param3 then we write the pid number in this filename  //  // Parameters  // ----------  // $1 (Text) Appli name to find  // $2 (Pointeur) pid number returned  // $3 (Text) name of the pidfilename (opt)  //  // Pgm Extern  // ----------  // pid_find_write.sh in resources/bash  // ----------------------------------------------------C_TEXT($1;$AppName;$3;$PidFileName)C_TEXT($Action;$In;$Out;$Error)C_POINTER($2)$0:=""If (Count parameters>=2)$AppName:="\""+$1+"\""$respath:=Get 4D folder(Current Resources folder)+"bash:"SET ENVIRONMENT VARIABLE("_4D_OPTION_CURRENT_DIRECTORY";$respath)Case of : (Count parameters=3)  // Write pid file in $3If ($3#"")$PidFileName:=$3$Action:="./pid_find_write.sh "+$AppName+" ./"+$PidFileNameElse $0:="Erreur de parametres"End if : (Count parameters=2)$Action:="./pid_find.sh "+$AppNameElse $0:="Erreur de paramtres"End case If ($0="")LAUNCH EXTERNAL PROCESS($Action;$In;$Out;$Error)If ($Error="")$2->:=Util_Trim_After ($Out)  // remove blank and \n \tEnd if $0:=$ErrorEnd if Else $0:="Erreur de paramtres"End if
